each-loops

  
    // Set up your $colors map
    $colors: (
      'color-primary': #e7ad10,
      'color-secondary': #d8531e,
      'color-tertiary': #747501
    );

    // Then loop through each color name and hex
    @each $color-name, $hex in $colors {
      .t-#{$color-name} {
        color: $hex;
      }
    }
  

  
    // Outputted CSS
    .t-color-primary {
      color: #e7ad10;
    }

    .t-color-secondary {
      color: #d8531e;
    }

    .t-color-tertiary {
      color: #747501;
    }
