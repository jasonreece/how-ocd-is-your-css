sass-variables

  
    // Easily undestandable variables
    $color-yellow: #e7ad10;
    $color-orange: #d8531e;
    $color-green: #747501;

    // Set to more scalable variables
    $color-primary: $color-yellow;
    $color-secondary: $color-orange;
    $color-tertiary: $color-green;

    // Using the variable
    .t-color-primary {
      color: $color-primary;
    }
  

  
    // Outputted CSS
    .t-color-primary {
      color: #e7ad10;
    }
