bad-nesting

  • Follow the inception rule.
  
    // Beware of specificity
    // from nesting too deeply!
    nav {
      ul {
        li {
          float: left;
          a {
            color: blue;
            &:hover,
            &:focus,
            &:active {
              color: red;
            }
          }
        }
      }
    }
  

  
    // Outputted CSS
    nav ul li {
      float: left;
    }

    nav ul li a {
      color: blue;
    }

    nav ul li a:hover,
    nav ul li a:focus,
    nav ul li a:active {
      color: red;
    }
