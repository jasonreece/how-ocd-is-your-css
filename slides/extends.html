sass-extends

  
    // Set up the @extend
    .center {
      display: block;
      margin-left: auto;
      margin-right: auto;
    }

    // Using the @extend
    .container-one {
      @extend .center;
      background: red;
    }

    .container-two {
      @extend .center;
      background: blue;
    }
  

  
    // Outputted CSS
    // Better than the mixin, but still not DRY.
    .center, .container-one, .container-two {
      display: block;
      margin-left: auto;
      margin-right: auto;
    }

    .container-one {
      background: red;
    }

    .container-two {
      background: blue;
    }
