sass-functions

  
    // Set up the responsive width function
    @function rw($target, $context) {
      @return ($target / $context) * 100%;
    }

    // Using the rw function
    .container-one {
      width: rw(720, 960);
    }

    .container-two {
      width: rw(240, 960);
    }
  

  
    // Outputted CSS
    .container-one {
      width: 75%;
    }

    .container-two {
      width: 25%;
    }
