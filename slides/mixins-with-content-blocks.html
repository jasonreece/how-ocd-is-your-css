sass-content-blocks

    
      // Adds responsive breakpoints.
      @mixin breakpoint($width) {
        @media (min-width: $width) {
          @content;
        }
      }

      // Nest breakpoints in rulesets
      .l-container {
        width: 90%;
        @include breakpoint(768px) {
          width: 50%;
        }
      }
    
  
    
      // Outputted CSS
      .l-container {
        width: 90%;
      }

      @media (min-width: 768px) {
        .l-container {
          width: 50%;
        }
      }

sass-content-blocks

    
      // Another good use for @content
      @mixin attention() {
        &:hover,
        &:active,
        &:focus {
          @content;
        }
      }

      a {
        color: red;
        @include attention() {
          color: blue;
        }
      }
    
  
    
      // Outputted CSS
      a {
        color: red;
      }

      a:hover, a:active, a:focus {
        color: blue;
      }
