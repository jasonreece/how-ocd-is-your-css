for-loops

  
    // Create a 12 column grid
    @for $i from 1 through 12 {
      %l-grid-#{$i} {
        width: ($i/12) * 100%;
      }
    }

    // Extend the placeholders
    .l-container-left {
      @extend %l-grid-9;  // (9/12) * 100% = 75%
    }

    .l-container-right {
      @extend %l-grid-3;  // (3/12) * 100% = 25%
    }
  

  
    // Outputted CSS
    .l-container-left {
      width: 75%;
    }

    .l-container-right {
      width: 25%;
    }
